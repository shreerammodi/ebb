# Groups Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Let a debater bundle adjacent cells in a column into a labeled **group** (a red/blue bracket), bundle-toggle them, and answer the whole group with one response — an additive overlay on the live argument tree.

**Architecture:** A `Round.groups: ArgGroup[]` collection, each `{ id, sheetId, label, memberIds[] }`, where members are nodes in the same column. Pure ops in `model/groups.ts`; undoable store actions via `_commit`; a `GroupBracket` styled overlay in `FlowGrid`; respond-to-group routes `node.answerAcross` to the group's anchor (first member). Layout/`buildLayout` is untouched — groups never change tree structure.

**Tech Stack:** Next.js + React, Zustand, TypeScript, Vitest + Testing Library, Dexie. Blue = Aff, red = Neg, light mode.

**Prerequisite:** Plan `2026-06-04-cell-display-editing-overhaul.md` is merged (this plan assumes `bold` + per-sheet columns exist). **Order:** run this before or after Plan 3 (cross-apps); both are additive. If both touch `normalize.ts` / `createRound`, keep each addition on its own line.

**Baseline:** Run `npm test -- --run` first and record the green count; keep it green each task.

import type { ArgGroup } from "@/lib/model/types";
import { uid } from "@/lib/model/ids";

export function groupForNode(groups: ArgGroup[], nodeId: string): ArgGroup | null {
  return groups.find((g) => g.memberIds.includes(nodeId)) ?? null;
}

export function createGroup(
  groups: ArgGroup[],
  input: { sheetId: string; memberIds: string[]; label?: string },
): ArgGroup[] {
  const group: ArgGroup = {
    id: uid("group"),
    sheetId: input.sheetId,
    label: input.label ?? "",
    memberIds: [...input.memberIds],
  };
  return [...groups, group];
}

export function addMember(groups: ArgGroup[], groupId: string, nodeId: string): ArgGroup[] {
  return groups.map((g) =>
    g.id === groupId && !g.memberIds.includes(nodeId)
      ? { ...g, memberIds: [...g.memberIds, nodeId] }
      : g,
  );
}

/** Removes a node from whatever group holds it; dissolves the group if <2 remain. */
export function removeMemberOrDelete(groups: ArgGroup[], nodeId: string): ArgGroup[] {
  return groups
    .map((g) =>
      g.memberIds.includes(nodeId)
        ? { ...g, memberIds: g.memberIds.filter((id) => id !== nodeId) }
        : g,
    )
    .filter((g) => g.memberIds.length >= 2);
}

export function setLabel(groups: ArgGroup[], groupId: string, label: string): ArgGroup[] {
  return groups.map((g) => (g.id === groupId ? { ...g, label } : g));
}
```

  // ── Groups ─────────────────────────────────────────────────────────────────
  groupNodes(sheetId, nodeIds, label) {
    if (!get().round || nodeIds.length < 2) return;
    get()._commit(null, (r) => ({ ...r, groups: createGroup(r.groups, { sheetId, memberIds: nodeIds, label }) }));
  },
  addToGroup(groupId, nodeId) {
    if (!get().round) return;
    get()._commit(null, (r) => ({ ...r, groups: addMember(r.groups, groupId, nodeId) }));
  },
  ungroupNode(nodeId) {
    if (!get().round) return;
    get()._commit(null, (r) => ({ ...r, groups: removeMemberOrDelete(r.groups, nodeId) }));
  },
  setGroupLabel(groupId, label) {
    if (!get().round) return;
    get()._commit(`grouplabel:${groupId}`, (r) => ({ ...r, groups: setGroupLabelOp(r.groups, groupId, label) }));
  },
```
